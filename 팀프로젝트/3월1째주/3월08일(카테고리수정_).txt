create table exr_notice(
	exr_notice_idx int primary key auto_increment
	,exr_category_idx int
	,title varchar(30)
	,writer varchar(30)
	,content text
	,regdate timestamp default now()
	,constraint fk_exr_category_notice foreign key (exr_category_idx) references exr_category (exr_category_idx)
)default character set utf8;
