create table exr_routine_comment(
	exr_routine_comment_idx int primary key auto_increment
	, exr_routine_idx int
	, member_idx int
	, title varchar(100)
	, writer varchar(30)
	, content mediumtext
	, regdate timestamp default now()
	, post int default 0
	, step int default 0
	, depth int default 0
	, constraint fk_exr_routine_comment foreign key (exr_routine_idx) references exr_routine (exr_routine_idx)
	, constraint fk_exr_member_routine_comment foreign key (member_idx) references member (member_idx)
)default character set utf8;
