create table exr_routine(
	exr_routine_idx int primary key auto_increment
	, exr_category_idx int
	, member_idx int
	, title varchar(100)
	, writer varchar(30)
	, content mediumtext
	, recommend int default 0
	, hit int default 0
	, regdate timestamp default now()
	, constraint fk_exr_category_routine foreign key (exr_category_idx) references exr_category (exr_category_idx)
	, constraint fk_exr_member_routine foreign key (member_idx) references member (member_idx)
)default character set utf8;
