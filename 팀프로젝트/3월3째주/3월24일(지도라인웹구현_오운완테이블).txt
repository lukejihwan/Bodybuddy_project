// 오운완 게시판


create table exr_today(
	exr_today_idx int primary key auto_increment
	, member_idx int
	, title varchar(100)
	, writer varchar(30)
	, content mediumtext
	, regdate timestamp default now()
	, hit int default 0
	, recommend int default 0
	, constraint fk_exr_member_today foreign key (member_idx) references member (member_idx)
)default character set utf8;

// 오운완 댓글 테이블
create table exr_today_comment(
	exr_today_comment_idx int primary key auto_increment
	, exr_today_idx int
	, member_idx int
	, title varchar(100)
	, writer varchar(30)
	, content text
	, regdate timestamp default now()
	, post int default 0
	, step int default 0
	, depth int default 0
	, constraint fk_exr_today_comment foreign key (exr_today_idx) references exr_today (exr_today_idx)
	, constraint fk_member_exr_today_comment foreign key (member_idx) references member (member_idx)
)default character set utf8;
