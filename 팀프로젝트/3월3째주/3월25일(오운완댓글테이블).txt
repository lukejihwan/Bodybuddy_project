create table exr_today_comment(
	exr_today_comment_idx int primary key auto_increment
	, exr_today_idx int
	, member_idx int
	, writer varchar(30)
	, content text
	, regdate timestamp default now()
	, post int default 0
	, step int default 0
	, depth int default 0
	, constraint fk_exr_today_comment foreign key (exr_today_idx) references exr_today (exr_today_idx)
	, constraint fk_member_exr_today_comment foreign key (member_idx) references member (member_idx)
)default character set utf8;


alter table exr_today_comment drop column title;
